class Sorter:
    def __init__(self):
        pass

    def sort(self, nums: list[int]) -> None:
        """Sorts nums in ascending order in place"""
        if len(nums) == 0:
            return
        else:
            self.sort(nums[1:])
            self.insert(nums, 0, nums[0])
            return nums
        
    def insert(self, nums: list[int], hi: int, num: int) -> None:
        """Inserts the given number at the correct spot in the section of nums to the left of hi"""
        for i in range(hi, -1, -1):
            if nums[i] > num:
                nums[i + 1] = nums[i]
            else:
                nums[i + 1] = num
                break
        else:
            nums[0] = num
