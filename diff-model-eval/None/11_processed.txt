class Fib:
    def __iter__(self):
        self.prev_prev = 0
        self.prev = 1
        return self
    def __next__(self):
        output = self.prev + self.prev_prev
        self.prev_prev = self.prev
        self.prev = output
        return output

    def next_n_fibs(self, n):
        fibs = []
        fib_iter = iter(self)
        for _ in range(n):
            fibs.append(next(fib_iter))
        return fibs
