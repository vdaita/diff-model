from typing import List


def binary_search(lst: List[int], x: int) -> int:
    low = 0
    high = len(lst) - 1
    mid = 0
    while low <= high:
        mid = (high + low) // 2
        if lst[mid] < x:
            low = mid + 1
        elif lst[mid] > x:
            high = mid - 1
        else:
            if mid == 0 or lst[mid-1] != x:
                return mid
            high = mid - 1
    return -1
