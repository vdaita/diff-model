from typing import List, Tuple

def max_subarray_sum(lst: List[int]) -> Tuple[int, int, int]:
    if not lst: return 0, 0, 0
    max_so_far = lst[0]
    curr_max = lst[0]
    srt_idx = 0
    end_idx = 0
    for i in range(1, len(lst)):
        if curr_max <= 0:
            curr_max = lst[i]
            srt_idx = i
        else:
            curr_max += lst[i]
            end_idx = i
        if curr_max > max_so_far:
            max_so_far = curr_max
    return max_so_far, srt_idx, end_idx
